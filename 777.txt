-- 1(a)
SELECT e.*, s.salary
FROM employees e
INNER JOIN salaries s
    ON e.emp_no = s.emp_no
WHERE s.salary = (
    SELECT MAX(salary)
    FROM salaries
);

-- 1(b)
SELECT e.*, d.*
FROM employees e
INNER JOIN dept_manager dm
    ON e.emp_no = dm.emp_no
INNER JOIN departments d
    ON dm.dept_no = d.dept_no
WHERE d.dept_no = 'd001'
  AND dm.to_date = '9999-01-01';

-- 1(c)
SELECT d1.*, d2.cnt
FROM departments d1
INNER JOIN (
    SELECT dept_no, COUNT(emp_no) AS cnt
    FROM dept_emp
    GROUP BY dept_no
) AS d2
    ON d1.dept_no = d2.dept_no
ORDER BY cnt DESC;

-- 2(b)
SELECT e.*, t.title
FROM employees e
INNER JOIN titles t
    ON e.emp_no = t.emp_no
WHERE t.title LIKE '%Engineer%'
  AND NOT EXISTS (
      SELECT 1
      FROM titles t2
      WHERE t2.emp_no = e.emp_no
        AND t2.title NOT LIKE '%Engineer%'
  );

-- 2(c)
SELECT e.*
FROM employees e
WHERE EXISTS (
    SELECT 1
    FROM dept_emp de
    WHERE de.emp_no = e.emp_no
      AND de.dept_no = 'd003'
)
AND EXISTS (
    SELECT 1
    FROM adr_emp ae
    INNER JOIN address a
        ON ae.adr_no = a.adr_no
    WHERE ae.emp_no = e.emp_no
      AND ae.adr_type = 'P'
      AND a.country = 'Austria'
);

SELECT DISTINCT e.*
FROM employees e
INNER JOIN dept_emp de
    ON e.emp_no = de.emp_no
INNER JOIN adr_emp ae
    ON e.emp_no = ae.emp_no
INNER JOIN address a
    ON ae.adr_no = a.adr_no
WHERE de.dept_no = 'd003'
  AND a.country = 'Austria'
  AND ae.adr_type = 'T';
