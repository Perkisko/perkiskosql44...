SELECT DISTINCT e.*
FROM employees e
INNER JOIN adr_emp ae
    ON e.emp_no = ae.emp_no
INNER JOIN address a
    ON ae.adr_no = a.adr_no
WHERE a.country = 'Sierra Leone'
  AND ae.adr_type = 'T'
  AND EXISTS (
      SELECT 1
      FROM dept_emp de1
      WHERE de1.emp_no = e.emp_no
        AND de1.dept_no = 'd004'
  )
  AND EXISTS (
      SELECT 1
      FROM dept_emp de2
      WHERE de2.emp_no = e.emp_no
        AND de2.dept_no = 'd006'
  );

MAX(floor(datediff(curdate(),birth_date)/365)) as max_age,

AVG(FLOOR(DATEDIFF(CURDATE(), hire_date) / 365)) AS avg_years,
MIN(FLOOR(DATEDIFF(CURDATE(), hire_date) / 365)) AS min_years,
MAX(FLOOR(DATEDIFF(CURDATE(), hire_date) / 365)) AS max_years
FROM employees;

FROM salaries
WHERE salary = (
    SELECT MAX(salary)
    FROM salaries
);

FROM salaries
WHERE salary = (
    SELECT MIN(salary)
    FROM salaries
);
